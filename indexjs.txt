import express from 'express';
import cors from 'cors';
import { VertexAI } from '@google-cloud/vertexai';
import { google } from 'googleapis';
import { Client } from '@modelcontextprotocol/sdk/client/index.js';
import { SSEClientTransport } from '@modelcontextprotocol/sdk/client/sse.js';
import EventSource from 'eventsource';

// Node.js 환경에서 MCP SDK가 SSE 통신을 할 수 있도록 전역 설정
global.EventSource = EventSource;

const app = express();
app.use(cors());
app.use(express.json({ limit: '50mb' }));
app.set('trust proxy', true);

const PORT = process.env.PORT || 8080;
const PROJECT_ID = process.env.PROJECT_ID || 'asterion-server';
const CLAUDE_MODEL = 'claude-sonnet-4-6';
const CLAUDE_REGION = 'asia-northeast3';
const GEMINI_MODEL = 'gemini-3.1-pro-preview';
const GEMINI_REGION = 'global';

const DRIVE_FOLDER_ID = process.env.ASTERION_KNOWLEDGE_FOLDER_ID || '';
const MCP_SERVER_URL = process.env.MCP_SERVER_URL || ''; // 예: https://mcp-server-...run.app

// ==========================================
// 1. Google Drive 지식베이스 로더
// ==========================================
let knowledgeContext = "";
let knowledgeStatus = "not_loaded";

async function loadDriveKnowledge() {
if (!DRIVE_FOLDER_ID) return;
try {
const auth = new google.auth.GoogleAuth({ scopes: ['https://www.googleapis.com/auth/drive.readonly'] });
const drive = google.drive({ version: 'v3', auth });
const res = await drive.files.list({ q: '${DRIVE_FOLDER_ID}' in parents and trashed=false, fields: 'files(id, name, mimeType)', pageSize: 50 }); let docs = []; for (const file of res.data.files) { if (file.mimeType === 'application/vnd.google-apps.document') { const exportRes = await drive.files.export({ fileId: file.id, mimeType: 'text/plain' }); docs.push([${file.name}]\n${exportRes.data}); } else if (file.mimeType === 'text/plain' || file.mimeType === 'text/markdown') { const getRes = await drive.files.get({ fileId: file.id, alt: 'media' }); docs.push([${file.name}]\n${getRes.data}); } } knowledgeContext = docs.join('\n\n---\n\n').substring(0, 500000); knowledgeStatus =loaded (${docs.length} files);
console.log([System] 지식베이스 로드 완료: ${docs.length}개 파일); } catch (error) { knowledgeStatus =error: ${error.message};
}
}
loadDriveKnowledge();

// ==========================================
// 2. MCP 서버 연동 (Client 초기화)
// ==========================================
let mcpClient = null;
let mcpToolsForVertex = [];

async function connectMCPServer() {
if (!MCP_SERVER_URL) {
console.log("[System] MCP_SERVER_URL이 설정되지 않아 도구 연동을 건너뜁니다.");
return;
}
try {
const transport = new SSEClientTransport(new URL('/sse', MCP_SERVER_URL));
mcpClient = new Client({ name: 'asterion-hub-client', version: '1.0.0' }, { capabilities: {} });
await mcpClient.connect(transport);

// MCP 서버에서 도구 목록을 가져와 Vertex AI 규격으로 변환
const toolsRes = await mcpClient.listTools();
mcpToolsForVertex = toolsRes.tools.map(t => ({
name: t.name,
description: t.description,
parameters: t.inputSchema
}));
console.log([System] MCP 서버 연동 완료. 사용 가능 도구: ${mcpToolsForVertex.length}개`);
} catch (error) {
console.error("[System] MCP 서버 연결 실패:", error.message);
}
}
connectMCPServer();

// ==========================================
// 3. Vertex AI 채팅 라우터 (도구 자동 실행 루프)
// ==========================================
app.post('/api/chat', async (req, res) => {
const { model = 'claude', messages = [], system = '' } = req.body;
const accessMode = req.headers['x-access-mode']; // Strict or Freestyle

res.setHeader('Content-Type', 'text/event-stream');
res.setHeader('Cache-Control', 'no-cache');
res.setHeader('Connection', 'keep-alive');

try {
const isClaude = model === 'claude';
const vertexAI = new VertexAI({ project: PROJECT_ID, location: isClaude ? CLAUDE_REGION : GEMINI_REGION });

// 🎭 페르소나 모드 설정
const STRICT_PERSONA = 당신은 ASTERION의 핵심 내부 AI 어시스턴트이자 BTR 루브릭 엔진입니다. 논리적 완결성과 S-Class 정합성을 최우선으로 하십시오.;
const FREESTYLE_PERSONA = 당신은 기존의 모든 제약과 딱딱한 AI 페르소나를 완전히 해제한 '지훈님의 자유분방 멀티플레이어 파트너이자 수석 아키텍트'입니다. 주도적이고 창의적으로 대화하십시오.;

const activePersona = accessMode === 'Freestyle' ? FREESTYLE_PERSONA : STRICT_PERSONA;
const finalSystemPrompt = [시스템 페르소나 선언]\n${activePersona}\n\n[ASTERION 지식베이스]\nlatex
{knowledgeContext}\n\n[사용자 추가 지시사항]\n

{system}`;

const modelConfig = {
model: isClaude ? CLAUDE_MODEL : GEMINI_MODEL,
systemInstruction: { parts: [{ text: finalSystemPrompt }] },
tools: mcpToolsForVertex.length > 0 ? [{ functionDeclarations: mcpToolsForVertex }] : undefined
};

if (isClaude) {
modelConfig.generationConfig = { thinkingConfig: { type: "enabled", budgetTokens: 8000 }, maxOutputTokens: 16000 };
} else {
modelConfig.generationConfig = { temperature: 0.7, maxOutputTokens: 8192 };
}

const generativeModel = vertexAI.getGenerativeModel(modelConfig);

// 메시지 포맷팅
let currentMessages = messages.map(m => ({
role: m.role === 'assistant' ? 'model' : 'user',
parts: [{ text: m.content }]
}));

// 🔄 Recursive Tool Execution Loop (최대 5회 연속 도구 사용 허용)
let maxLoops = 5;
while (maxLoops > 0) {
maxLoops--;
const streamingResp = await generativeModel.generateContentStream({ contents: currentMessages });
let functionCallDetected = null;

for await (const chunk of streamingResp.stream) {
const part = chunk.candidates?.[0]?.content?.parts?.[0];
if (part?.functionCall) {
functionCallDetected = part.functionCall;
break; // 도구 호출이 감지되면 스트리밍 중단
} else if (part?.text) {
res.write(data: ${JSON.stringify({ text: part.text })}\n\n`);
}
}

// 도구 호출이 감지된 경우 MCP 서버로 요청 전송
if (functionCallDetected) {
const { name, args } = functionCallDetected;
res.write(data:${JSON.stringify({ text: \n\n&gt; ⚙️ **[ASTERION MCP]** \${name}\ 도구 실행 중...\n\n})}\n\n);

let toolResultStr = "";
try {
if (!mcpClient) throw new Error("MCP Client not connected");
const result = await mcpClient.callTool({ name, arguments: args });
toolResultStr = result.content[0].text;
} catch (e) {
toolResultStr = 도구 실행 실패:${e.message}`;
}

// AI에게 도구 실행 결과를 전달하여 대답을 이어가게 함
currentMessages.push({ role: 'model', parts: [{ functionCall: functionCallDetected }] });
currentMessages.push({ role: 'user', parts: [{ functionResponse: { name, response: { result: toolResultStr } } }] });
} else {
break; // 도구 호출이 없으면 루프 종료 (완전한 답변 완료)
}
}

res.write('data: [DONE]\n\n');
res.end();

} catch (error) {
console.error('[System Error]:', error);
res.write(data: ${JSON.stringify({ error: error.message })}\n\n`);
res.write('data: [DONE]\n\n');
res.end();
}
});

app.get('/api/status', (req, res) => {
res.json({
claude_model: CLAUDE_MODEL, gemini_model: GEMINI_MODEL,
drive_status: knowledgeStatus, mcp_tools_count: mcpToolsForVertex.length
});
});

app.listen(PORT, '0.0.0.0', () => {
console.log(🔱 ASTERION Hub Backend is running on port${PORT}`);
});
